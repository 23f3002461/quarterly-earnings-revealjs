Data Pipeline (Python)


import pandas as pd

def compute_qoq_growth(df):
    df = df.sort_values(['region','quarter'])
    df['rev_lag'] = df.groupby('region').revenue.shift(1)
    df['qoq_growth'] = (df.revenue - df.rev_lag) / df.rev_lag
    return df
